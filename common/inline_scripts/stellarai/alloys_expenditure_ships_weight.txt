weight = {
		weight = 0.6

		inline_script = {
			script = stellarai/early_game_navy_weight_modifiers
		}

		modifier = {
			factor = 8
			AND = {
				years_passed < 41
				years_passed > 5
				used_naval_capacity_percent < 0.4
			}
		}

		modifier = {
			factor = 8
			AND = {
				years_passed < 41
				years_passed > 20
				used_naval_capacity_percent < 0.8
			}
		}

		# spend more alloys building ships while at war
		modifier = {
			factor = 2
			OR = {
				recently_lost_war = yes
				is_at_war = yes
			}
		}

		modifier = {
			factor = 0.5
			AND = {
				ai_colonize_plans > 0
				is_at_war = no
				years_passed < 20
			}
		}

		# If we want to colonize and are at peace then prio that
		modifier = {
			factor = 0.25
			AND = {
				ai_colonize_plans > 0
				is_at_war = no
			}
		}

		# spend less alloys on ships if over capacity
		modifier = {
			factor = 0.5
			used_naval_capacity_percent >= 1.0
		}

		# Biological Ships use less alloys
		modifier = {
			factor = 0.33
			country_uses_bio_ships = yes
		}

		modifier = {
			factor = 5
			used_naval_capacity_percent < 1.0
			OR = {
				any_neighbor_country = {
					has_ascension_perk = ap_become_the_crisis
				}
				any_country = {
					is_crisis_faction = yes
				}
			}
		}
