# Ships should default to M slots, which are used primarily for lasers, mass drivers or plasma.
# We make no check for autocannons because they're fine with all sizes and also make good companion weapons for torpedoes.
# We make no check for X weapons because they're always the best and lacking resources doesn't stop the AI from including them in battleship designs.


P_is_good = { # P vs S&M
	OR = {
		has_technology = tech_battleships # After we have battleships, smaller ships should be used for escort duty.
		AND = { # energy & kinetic block
			NOR = {
				is_preferred_weapons = weapon_type_explosive
				is_preferred_weapons = weapon_type_strike_craft
			}
			OR = {
				AND = { # top tier PD, no top tier guns
					OR = {
						has_technology = tech_pd_tracking_3
						has_technology = tech_flak_batteries_3
					}
					NOR = {
						has_technology = tech_lasers_5
						has_technology = tech_mass_drivers_5
						has_technology = tech_plasma_3
        		        has_technology = tech_autocannons_3
					}
				}
				AND = { # mid tier PD, no mid tier guns
					OR = {
						has_technology = tech_pd_tracking_2
						has_technology = tech_flak_batteries_2
					}
					NOR = {
						has_technology = tech_lasers_3
						has_technology = tech_mass_drivers_3
						has_technology = tech_plasma_1
        		        has_technology = tech_autocannons_1
					}
				}
			}
		} # end of energy & kinetic block
		AND = { # missile & strike craft block
			OR = {
				is_preferred_weapons = weapon_type_explosive
				is_preferred_weapons = weapon_type_strike_craft
			}
			OR = {
				AND = { # top tier PD, no top tier missiles
					OR = {
						has_technology = tech_pd_tracking_3
						has_technology = tech_flak_batteries_3
					}
					NOR = {
						has_technology = tech_missiles_5
						has_technology = tech_swarmer_missiles_2
        		        has_technology = tech_disruptors_3
					}
				}
				AND = { # mid tier PD, no mid tier missiles
					OR = {
						has_technology = tech_pd_tracking_2
						has_technology = tech_flak_batteries_2
					}
					NOR = {
						has_technology = tech_missiles_3
						has_technology = tech_swarmer_missiles_1
        		        has_technology = tech_disruptors_1
					}
				}
			}
		} # end of missile & strike craft block
	} # end of first OR
}

M_is_good = { # M vs S
	OR = {
		AND = {
			NOR = { # We put this check on top because eventually it'll return false on the autocannons line and we won't need to check the rest.
				has_technology = tech_autocannons_1
				has_technology = tech_plasma_1
			}
			OR = {
				is_preferred_weapons = weapon_type_energy
				is_preferred_weapons = weapon_type_kinetic
			}
		}
		AND = {
			is_preferred_weapons = weapon_type_explosive
			OR = {
				has_technology = tech_swarmer_missiles_2
				AND = {
					NOT = { has_technology = tech_missiles_4 }
					has_technology = tech_swarmer_missiles_1
				}
			}
		}
		AND = {
			is_preferred_weapons = weapon_type_strike_craft
			has_technology = tech_swarmer_missiles_2
		}
	}
}

L_is_good = { # L vs M
	OR = {
		is_preferred_weapons = weapon_type_energy
		is_preferred_weapons = weapon_type_kinetic
	}
	NOT = { no_resource_for_component = { RESOURCE = volatile_motes } }
	has_technology = tech_kinetic_artillery_1
}

G_is_good = { # G vs everything
	has_technology = tech_battleships # If we have battleships, the enemy should at least have cruisers.
	OR = {
		AND = {
			is_preferred_weapons = weapon_type_explosive
			NOT = { no_resource_for_component = { RESOURCE = volatile_motes } }
			has_technology = tech_torpedoes_2
		}
		AND = {
			NOR = {
				is_preferred_weapons = weapon_type_explosive
				is_preferred_weapons = weapon_type_strike_craft
			}
			NOT = { no_resource_for_component = { RESOURCE = exotic_gases } }
			has_technology = tech_energy_torpedoes_1
		}
	}
}

has_x_weapon = {
	OR = {
		AND = {
			NOT = { no_resource_for_component = { RESOURCE = rare_crystals } }
			OR = {
				has_technology = tech_arc_emitter_1
				has_technology = tech_arc_emitter_2
				has_technology = tech_energy_lance_1
				has_technology = tech_energy_lance_2
			}
		}
		AND = {
			NOT = { no_resource_for_component = { RESOURCE = volatile_motes } }
			OR = {
				has_technology = tech_mass_accelerator_1
				has_technology = tech_mass_accelerator_2
			}
		}
		AND = {
			NOT = { no_resource_for_component = { RESOURCE = minor_artifacts } }
			has_technology = tech_archaeo_mass_accelerator
		}
	}
}

is_cruiser_user = {
	OR = {
		is_preferred_weapons = weapon_type_explosive
		is_preferred_weapons = weapon_type_strike_craft
		NOT = { has_technology = tech_battleships }
	}
	has_technology = tech_cruisers
	NOT = { has_ascension_perk = ap_armored_in_firepower }
}
