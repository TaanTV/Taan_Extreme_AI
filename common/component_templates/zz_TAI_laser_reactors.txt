# 1. Деление (Fission) - Начальный уровень
utility_component_template = {
	key = "CORVETTE_LASER_TAI_FISSION_REACTOR" # Уникальное имя!
	size = small
	icon = "GFX_ship_part_reactor_1"
	icon_frame = 1
	power = @corvette_reactor_power_1
	resources = {
		category = ship_components
		cost = {
			alloys = @corvette_reactor_cost_1
		}
		upkeep = {
			energy = 0.10
			alloys = 0.01
		}
	}

	prerequisites = { "tech_fission_power" }
	component_set = "power_core" # Это связывает компонент с слотом реактора
	
	# ВОТ ГЛАВНОЕ ИЗМЕНЕНИЕ:
	potential = {
		OR = {
			ship_uses_corvette_reactors = yes # Оставляем для обычных корветов
			is_ship_size = corvette_laser_TAI # Разрешаем для ТВОЕГО корабля
		}
		from = {
			country_uses_bio_ships = no
		}
	}

	show_tech_unlock_if = {
		country_uses_bio_ships = no
	}
	
	# Ссылка на следующий уровень (указываем на наш новый реактор)
	upgrades_to = "CORVETTE_LASER_TAI_FUSION_REACTOR"

	ai_weight = {
		weight = 1
	}
}
