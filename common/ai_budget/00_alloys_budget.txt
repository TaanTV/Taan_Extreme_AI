alloys_expenditure_ships = {
	resource = alloys
	type = expenditure
	category = ships

	potential = {
		always = yes
	}

	weight = {
		weight = 0.8

		# inline_script = {
		# 	script = stellarai/early_game_navy_weight_modifiers
		# }

		modifier = {
			factor = 2.5
			AND = {
				years_passed > 18
				years_passed < 41
				used_naval_capacity_percent < 0.9
			}
		}

		modifier = {
			factor = 2.5
			AND = {
				NOT = {
					has_technology = tech_cruisers
				}
				has_technology = tech_destroyers
				used_naval_capacity_percent < 1
				has_monthly_income = {
					resource = alloys
					value > 9
				}
			}
		}

		modifier = {
			factor = 3.5
			AND = {
				NOT = {
					has_technology = tech_battleships
				}
				has_technology = tech_cruisers
				used_naval_capacity_percent < 1
				has_monthly_income = {
					resource = alloys
					value > 20
				}
			}
		}

		modifier = {
			factor = 4.5
			AND = {
				has_technology = tech_battleships
				used_naval_capacity_percent < 1
				has_monthly_income = {
					resource = alloys
					value > 28
				}
			}
		}

		# spend more alloys building ships while at war
		modifier = {
			factor = 3
			OR = {
				recently_lost_war = yes
				is_at_war = yes
			}
		}

		# Strong emergency response under heavy military pressure
		modifier = {
			factor = 2.25
			stellarai_military_emergency = yes
		}

		# If we want to colonize and are at peace then prio that (less so for machines), unless under heavy military pressure
		modifier = {
			factor = 0.25
			AND = {
				ai_colonize_plans > 0
				is_at_war = no
				is_machine_empire = no
				NOT = { stellarai_military_emergency = yes }
			}
		}

		modifier = {
			factor = 0.50
			AND = {
				ai_colonize_plans > 0
				is_at_war = no
				is_machine_empire = yes
				NOT = { has_valid_civic = civic_machine_terminator }
				NOT = { stellarai_military_emergency = yes }
			}
		}

		modifier = {
			factor = 0.75
			AND = {
				ai_colonize_plans > 0
				is_at_war = no
				has_valid_civic = civic_machine_terminator
				NOT = { stellarai_military_emergency = yes }
			}
		}

		# spend less alloys on ships if over capacity
		modifier = {
			factor = 0.75
			stellarai_machine_energy_emergency = yes
		}

		modifier = {
			factor = 0.85
			stellarai_machine_minerals_emergency = yes
		}

		modifier = {
			factor = 1.15
			stellarai_machine_basic_econ_stable = yes
			used_naval_capacity_percent < 1.0
		}

		modifier = {
			factor = 1.25
			stellarai_exterminator_war_ramp = yes
		}

		modifier = {
			factor = 0.5
			used_naval_capacity_percent >= 1.0
		}

		# Biological Ships use less alloys
		modifier = {
			factor = 0.33
			country_uses_bio_ships = yes
		}

		modifier = {
			factor = 5
			used_naval_capacity_percent < 1.5
			OR = {
				any_neighbor_country = {
					has_ascension_perk = ap_become_the_crisis
				}
				any_country = {
					is_crisis_faction = yes
				}
			}
		}
	
	}

	desired_min = {
		base = 0

		modifier = {
			add = 3
			country_uses_bio_ships = no
		}

		modifier = {
			factor = trigger:years_passed
		}
	}
}

alloys_expenditure_starbases = {
	resource = alloys
	type = expenditure
	category = starbases

	potential = {
		NOT = {
			is_country_type = fallen_empire
		}
	}

	weight = {
		weight = 0.50

		# spend more on starbase upgrades if at fleet cap
		modifier = {
			factor = 2
			used_naval_capacity_percent >= 1.0
		}

		# Reduce spending on claiming new systems if we have unclaimed planets
		modifier = {
			is_machine_empire = yes
			has_country_resource = { type = alloys amount < 450 }
			factor = 0.5
			ai_colonize_plans > 0
		}
		# Reduce spending on claiming new systems if we have unclaimed planets
		modifier = {
			is_machine_empire = no
			has_country_resource = { type = alloys amount < 250 }
			factor = 0.75
			ai_colonize_plans > 0
		}

		modifier = {
			factor = 1.20
			is_machine_empire = yes
		}

		modifier = {
			factor = 1.35
			has_valid_civic = civic_machine_terminator
		}

		modifier = {
			factor = 1.20
			stellarai_machine_expand_ok = yes
		}

		modifier = {
			factor = 1.75
			used_naval_capacity_percent > 0.75
			any_owned_starbase = {
				can_be_upgraded = yes
			}
		}

		modifier = {
			factor = 1.25
			highest_threat > 20
			OR = {
				any_owned_starbase = {
					can_be_upgraded = yes
				}
				any_owned_nonprimary_starbase = {
					can_be_upgraded = yes
				}
			}
		}

		modifier = {
			factor = 1.5
			OR = {
				any_owned_starbase = {
					NOT = { has_starbase_size = starbase_outpost }
					can_be_upgraded = yes
				}
				any_owned_nonprimary_starbase = {
					can_be_upgraded = yes
				}
			}
		}

		# Biological Starbases use less alloys
		modifier = {
			factor = 0.5
			country_uses_bio_ships = yes
		}
	}
	desired_max = {
		base = 1100
		modifier = {
			add = 5000
			has_ascension_perk = ap_cosmogenesis
		}
	}
}

alloys_expenditure_decisions = {

			}			
		}

			any_owned_planet = {
				is_planet_class = pc_shattered_ring_habitable

			}
		}
