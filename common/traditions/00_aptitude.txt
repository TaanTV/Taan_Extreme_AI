tr_aptitude_finish = {
	custom_tooltip_with_modifiers = tr_aptitude_adopt_desc

	on_enabled = {
		hidden_effect = {
			every_owned_leader = {
				add_trait = {
					trait = random_common
					show_message = yes
				}
			}
			every_pool_leader = {
				add_trait = {
					trait = random_common
					show_message = no
				}
			}
		}
	}

	triggered_modifier = {
		potential = {
			always = yes
		}
		ascension_perks_add = 1
	}
}
